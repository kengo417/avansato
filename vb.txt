select
    ses.USERNAME,
    to_char(seg.blocks * 8192 /1024/1024, '999,999.000') Mbytes,
    ses.SCHEMANAME,
    ses.OSUSER,
    ses.PROGRAM,
    sqla.SQL_FULLTEXT
from
    v$tempseg_usage seg
    inner join v$session ses on(seg.SESSION_ADDR=ses.SADDR)
    inner join v$sqlarea sqla on(seg.SQL_ID=sqla.SQL_ID)
order by 1,2 desc;
